        var expires = new Date();

        function processLogin(form){
                var status = new Boolean(validateForm(form));
                if (status == true){
                        //saveUsername(form);

                        document.amLogin.IDToken2.value = document.loginForm.j_password.value;

                                document.amLogin.iPSPCookie.value = 'yes';

                                if (document.amLogin.goto.value.indexOf('rm=1') &gt; -1) {}
                                else {
                                        var appendGoto = false;
                                        var separator = (document.amLogin.goto.value.indexOf('?') &gt; -1) ? "&amp;" : "?";

                                        if (document.amLogin.goto.value.indexOf('cdcservlet') &gt; -1) {
                                                var targetIndex = document.amLogin.goto.value.indexOf('TARGET');

                                                if (targetIndex &gt; -1) {
                                                        separator = (document.amLogin.goto.value.indexOf('%3F') &gt; -1) ? "%26" : "%3F";

                                                        var targetA = document.amLogin.goto.value.substring(0, targetIndex);
                                                        var targetB = document.amLogin.goto.value.substring(targetIndex);
                                                        var targetBIndex = targetB.indexOf('&amp;');

                                                        if (targetBIndex &gt; -1) {
                                                                var targetBa = targetB.substring(0, targetBIndex);
                                                                var targetBb = targetB.substring(targetBIndex);

                                                        }
